""" FinanceAI — CLI Entry Point Usage examples: # Full run with PDF directory + CSV files python main.py \ --invoices data/invoices/ \ --payments data/payments.csv \ --expenses data/expenses.csv \ --output data/reports/ledger.json # Single PDF python main.py --pdf path/to/invoice.pdf --payments data/payments.csv # Demo mode (no PDFs needed) python main.py --demo """ from __future__ import annotations import argparse import sys from pathlib import Path sys.path.insert(0, str(Path(__file__).parent)) import config from utils.logger import get_logger logger = get_logger("main") def build_parser() -> argparse.ArgumentParser: p = argparse.ArgumentParser( prog="financeai", description="FinanceAI — Automated invoice reconciliation agent.", ) p.add_argument( "--invoices", type=Path, metavar="DIR", help="Directory containing invoice PDFs.", ) p.add_argument( "--pdf", type=Path, nargs="+", metavar="FILE", help="One or more specific invoice PDF paths.", ) p.add_argument( "--payments", type=Path, metavar="CSV", help="Payment log CSV file.", ) p.add_argument( "--expenses", type=Path, metavar="CSV", help="Expense spreadsheet (CSV or XLSX).", ) p.add_argument( "--output", type=Path, default=config.LEDGER_OUTPUT_FILE, metavar="JSON", help=f"Output ledger path (default: {config.LEDGER_OUTPUT_FILE}).", ) p.add_argument( "--demo", action="store_true", help="Run demo mode with built-in sample data (no API key or PDFs needed).", ) return p def main() -> int: args = build_parser().parse_args() if args.demo: logger.info("Running in demo mode …") from demo import main as demo_main demo_main() return 0 if not args.invoices and not args.pdf: logger.error("Provide --invoices and/or --pdf …") return 1 if not args.payments: logger.error("Provide --payments ") return 1 if not config.ANTHROPIC_API_KEY: logger.error( "ANTHROPIC_API_KEY environment variable is not set. " "Export it before running: export ANTHROPIC_API_KEY=sk-ant-…" ) return 1 from agents.finance_agent import FinanceAgent agent = FinanceAgent() agent.load_invoices( invoice_dir=args.invoices, pdf_files =args.pdf or [], ) agent.load_payments(args.payments) if args.expenses: agent.load_expenses(args.expenses) agent.run(output_path=args.output) return 0 if __name__ == "__main__": sys.exit(main())
