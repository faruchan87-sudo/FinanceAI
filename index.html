""" from __future__ import annotations import sys from collections import defaultdict from datetime import date, timedelta from difflib import SequenceMatcher from pathlib import Path from typing import Dict, List, Set, Tuple sys.path.insert(0, str(Path(__file__).parent.parent)) import config from utils.logger import get_logger from utils.models import Invoice logger = get_logger("validators.duplicate_detector") _FUZZY_VENDOR_THRESHOLD = 0.85 # SequenceMatcher ratio; 1.0 = identical def _vendor_similarity(a: str, b: str) -> float: return SequenceMatcher(None, a.lower().strip(), b.lower().strip()).ratio() def _same_amount(a: float, b: float) -> bool: return abs(a - b) <= config.AMOUNT_TOLERANCE def _within_window(d1: date | None, d2: date | None) -> bool: if d1 is None or d2 is None: return True # unknown dates → conservatively flag return abs((d1 - d2).days) <= config.DUPLICATE_WINDOW_DAYS # ── Core logic ──────────────────────────────────────────────────────────────── class DuplicateDetector: """ Stateful detector: ingest invoices one at a time; call report() when done. """ def __init__(self) -> None: self._invoices: List[Invoice] = [] def add(self, invoice: Invoice) -> None: self._invoices.append(invoice) # ── strategies ──────────────────────────────────────────────────────────── def _find_exact_number_dupes(self) -> Set[Tuple[str, str]]: """ Return pairs of UIDs that share (vendor, invoice_number). """ seen: Dict[Tuple[str, str], str] = {} # key → first UID dupes: Set[Tuple[str, str]] = set() for inv in self._invoices: key = (inv.vendor.lower().strip(), inv.invoice_number.lower().strip()) if key in seen: pair = tuple(sorted([seen[key], inv.uid])) # type: ignore[arg-type] dupes.add(pair) # type: ignore[arg-type] logger.warning( "Amount-window duplicate: vendor=%r amount=%.2f " "dates=%s/%s UIDs %s vs %s", a.vendor, a.total, a.invoice_date, b.invoice_date, a.uid, b.uid, ) return dupes def _find_fuzzy_vendor_dupes(self) -> Set[Tuple[str, str]]: """Fuzzy vendor name + same amount + same date (catches OCR/typos).""" dupes: Set[Tuple[str, str]] = set() for i, a in enumerate(self._invoices): for b in self._invoices[i + 1:]: if a.vendor.lower().strip() == b.vendor.lower().strip(): continue # already caught above sim = _vendor_similarity(a.vendor, b.vendor) if sim < _FUZZY_VENDOR_THRESHOLD: continue if not _same_amount(a.total, b.total): continue if a.invoice_date != b.invoice_date: continue pair = tuple(sorted([a.uid, b.uid])) # type: ignore[arg-type] dupes.add(pair) # type: ignore[arg-type] logger.warning( "Fuzzy-vendor duplicate: %r ≈ %r (sim=%.2f) " "amount=%.2f date=%s UIDs %s vs %s", a.vendor, b.vendor, sim, a.total, a.invoice_date, a.uid, b.uid, ) return dupes # """ logger.info("Running duplicate detection on %d invoices …", len(self._invoices)) exact = self._find_exact_number_dupes() window = self._find_amount_window_dupes() fuzzy = self._find_fuzzy_vendor_dupes() # Build per-uid map uid_to_reasons: Dict[str, List[str]] = defaultdict(list) for uid_a, uid_b in exact: msg = f"Exact duplicate of invoice sharing same number/vendor (paired with {uid_b[:8]}…)" uid_to_reasons[uid_a].append(msg) uid_to_reasons[uid_b].append( f"Exact duplicate of invoice sharing same number/vendor (paired with {uid_a[:8]}…)" ) for uid_a, uid_b in window - exact: msg = f"Same vendor+amount within {config.DUPLICATE_WINDOW_DAYS}d window (paired with {uid_b[:8]}…)" uid_to_reasons[uid_a].append(msg) uid_to_reasons[uid_b].append( f"Same vendor+amount within {config.DUPLICATE_WINDOW_DAYS}d window (paired with {uid_a[:8]}…)" ) for uid_a, uid_b in fuzzy - exact - window: msg = f"Fuzzy-vendor + same amount/date (paired with {uid_b[:8]}…)" uid_to_reasons[uid_a].append(msg) uid_to_reasons[uid_b].append( f"Fuzzy-vendor + same amount/date (paired with {uid_a[:8]}…)" ) total_flagged = len(uid_to_reasons) if total_flagged: logger.warning("%d invoice(s) flagged as potential duplicates.", total_flagged) else: logger.info("No duplicate invoices detected.") return dict(uid_to_reasons) def detect_duplicates(invoices: List[Invoice]) -> Dict[str, List[str]]: """Convenience wrapper: build a detector, add all invoices, return results.""" detector = DuplicateDetector() for inv in invoices: detector.add(inv) return detector.find_duplicates() """ FinanceAI — Reconciler & Ledger Generator Matches invoices to payment records; produces the final JSON ledger.
